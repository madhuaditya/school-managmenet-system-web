# Insert Queue Item

```ts id="0r6r7t"
export async function addQueueItem(item) {
  db.execute(
    `
    INSERT INTO attendance_queue (
      id,
      request_id,
      class_id,
      attendance_date,
      payload,
      status,
      retry_count,
      created_at,
      updated_at
    )
    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
    `,
    [
      item.id,
      item.requestId,
      item.classId,
      item.date,
      JSON.stringify(item.payload),
      'PENDING',
      0,
      Date.now(),
      Date.now(),
    ]
  );
}
```

# Get Pending Items

```ts id="awjdg5"
export async function getPendingItems() {
  const result = db.execute(`
    SELECT *
    FROM attendance_queue
    WHERE status IN ('PENDING', 'FAILED')
    ORDER BY created_at ASC
    LIMIT 10
  `);

  return result.rows?._array || [];
}
```

---

# Update Status

```ts id="s5jlwm"
export async function updateQueueStatus(
  id,
  status
) {
  db.execute(
    `
    UPDATE attendance_queue
    SET status = ?,
        updated_at = ?
    WHERE id = ?
    `,
    [status, Date.now(), id]
  );
}
```

---

# Increment Retry

```ts id="7v30tb"
export async function incrementRetry(
  id,
  retryCount,
  nextRetryAt
) {
  db.execute(
    `
    UPDATE attendance_queue
    SET retry_count = ?,
        next_retry_at = ?,
        status = 'FAILED'
    WHERE id = ?
    `,
    [retryCount, nextRetryAt, id]
  );
}
```

```ts id="jlwm9s"
import 'react-native-get-random-values';
import { v4 as uuid } from 'uuid';

export async function submitAttendance(
  records,
  classId,
  date
) {
  const requestId = uuid();

  const payload = {
    requestId,
    classId,
    date,
    records,
  };

  await addQueueItem({
    id: requestId,
    requestId,
    classId,
    date,
    payload,
  });

  showToast('Attendance Saved');

  startAttendanceSync();
}
```

# Core Sync Engine

```ts id="7ff3ep"
let isSyncing = false;

export async function startAttendanceSync() {

  if (isSyncing) return;

  isSyncing = true;

  try {

    const pendingItems =
      await getPendingItems();

    for (const item of pendingItems) {

      try {

        await updateQueueStatus(
          item.id,
          'SYNCING'
        );

        await uploadAttendance(
          JSON.parse(item.payload)
        );

        await updateQueueStatus(
          item.id,
          'SYNCED'
        );

      } catch (error) {

        const retryCount =
          item.retry_count + 1;

        const delay =
          getRetryDelay(retryCount);

        await incrementRetry(
          item.id,
          retryCount,
          Date.now() + delay
        );
      }
    }

  } finally {

    isSyncing = false;
  }
}
```

# PHASE 6 — Retry Delay Logic

# retryPolicy.ts

```ts id="t9v9z6"
export function getRetryDelay(
  retryCount
) {
  const delays = [
    5000,
    15000,
    30000,
    60000,
    300000,
  ];

  return delays[
    retryCount - 1
  ] || 300000;
}
```

```ts id="5yw8na"
from 'react-native-background-fetch';

export function configureBackgroundSync() {

  BackgroundFetch.configure(
    {
      minimumFetchInterval: 15,
      stopOnTerminate: false,
      startOnBoot: true,
      enableHeadless: true,
    },

    async (taskId) => {

      await startAttendanceSync();

      BackgroundFetch.finish(taskId);
    },
  );
}
```

# PHASE 10 — App Startup Recovery

# VERY IMPORTANT

If app crashes during syncing:

```text id="hjywjw"
SYNCING state gets stuck forever
```

Fix:

On app startup:

```sql id="mvh7jo"
UPDATE attendance_queue
SET status = 'PENDING'
WHERE status = 'SYNCING'
```

Then restart sync engine.
